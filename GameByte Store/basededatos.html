Modelo Base de Datos

Diagrama relacional del sistema GameByte Store

Diagrama Entidad-Relación

Este diagrama representa la estructura de la base de datos del sistema GameByte Store. Muestra las tablas principales, sus campos y las relaciones entre ellas. La base de datos permite almacenar información de usuarios, credenciales y productos, garantizando la integridad de los datos y el correcto funcionamiento del sistema.


Código Gestión Base de Datos

CREATE DATABASE GameByteDB;
USE GameByteDB;

-- Tabla de Usuarios
CREATE TABLE tblUsers (
    idUser INT PRIMARY KEY,
    fullName VARCHAR(100) NOT NULL,
    email VARCHAR(60) NOT NULL,
    phone VARCHAR(10)
);

-- Tabla de Login
CREATE TABLE tblLogin (
    idLogin INT IDENTITY(1000,11) PRIMARY KEY,
    idUser INT,
    userName VARCHAR(40) UNIQUE,
    password_unique VARCHAR(100) NOT NULL,
    CONSTRAINT FK_Login_User FOREIGN KEY (idUser) REFERENCES tblUsers(idUser)
);

-- Tabla de Categorías
CREATE TABLE tblCategory (
    idCategory INT IDENTITY(1,1) PRIMARY KEY,
    nameCategory VARCHAR(40) NOT NULL
);

-- Tabla de Productos
CREATE TABLE tblProducts (
    idProduct INT IDENTITY(1111,11) PRIMARY KEY,
    nameProduct VARCHAR(100) NOT NULL,
    details VARCHAR(300) NOT NULL,
    price DECIMAL(10,2),
    stock INT,
    idCategory INT,
    CONSTRAINT FK_Products_Category FOREIGN KEY (idCategory) REFERENCES tblCategory(idCategory)
);

-- Tabla de Estados
CREATE TABLE tblStatus (
    idStatus INT PRIMARY KEY,
    statusName VARCHAR(30) NOT NULL
);

-- Tabla de Órdenes
CREATE TABLE tblOrder (
    idOrder INT IDENTITY(10000,11) PRIMARY KEY,
    idUser INT,
    idStatus INT,
    dateOrder DATE,
    totalOrder DECIMAL(10,2),
    addressOrder VARCHAR(100),
    city VARCHAR(50),
    CONSTRAINT FK_Order_Users FOREIGN KEY (idUser) REFERENCES tblUsers(idUser),
    CONSTRAINT FK_Order_Status FOREIGN KEY (idStatus) REFERENCES tblStatus(idStatus)
);

-- Tabla de Detalles de Orden
CREATE TABLE tblOrderDetails (
    idDetail INT IDENTITY(10000,1),
    idOrder INT,
    idProduct INT,
    units INT,
    unitPrice DECIMAL(10,2),
    lineTotal DECIMAL(10,2),
    CONSTRAINT PK_tblOrderDetails PRIMARY KEY (idDetail),
    CONSTRAINT FK_OrderDetails_Order FOREIGN KEY (idOrder) REFERENCES tblOrder(idOrder),
    CONSTRAINT FK_OrderDetails_Products FOREIGN KEY (idProduct) REFERENCES tblProducts(idProduct)
);
